class ATGNode:
    def __init__(self, node_id, node_type, chain, address):
        self.node_id = node_id
        self.node_type = node_type  # user, contract, relay
        self.chain = chain          # source, destination, offchain
        self.address = address

class ATGEdge:
    def __init__(self, src, dst, label, conditions):
        self.src = src
        self.dst = dst
        self.label = label  # lock, unlock, mint, burn, relay, verify
        self.conditions = conditions  # list of guard predicates

class ATG:
    def __init__(self):
        self.nodes = {}
        self.edges = []
        self.invariants = []

    def build_from_llm_output(self, llm_json):
        # Parse entities -> nodes
        for entity in llm_json["entities"]:
            node = ATGNode(
                node_id=entity["id"],
                node_type=entity["type"],
                chain=entity["chain"],
                address=entity["address"]
            )
            self.nodes[node.node_id] = node

        # Parse asset_flows -> edges
        for flow in llm_json["asset_flows"]:
            edge = ATGEdge(
                src=flow["from"],
                dst=flow["to"],
                label=flow["operation"],
                conditions=flow.get("guards", [])
            )
            self.edges.append(edge)

# src/module2_rag/embedder.py

from sentence_transformers import SentenceTransformer
import faiss
import json
import numpy as np

class ExploitKnowledgeBase:
    def __init__(self, data_dir="src/module2_rag/data/"):
        self.model = SentenceTransformer('all-MiniLM-L6-v2')
        self.exploits = self._load_exploits(data_dir)
        self.index, self.embeddings = self._build_index()

    def _load_exploits(self, data_dir):
        exploits = []
        for filepath in sorted(Path(data_dir).glob("*.json")):
            with open(filepath) as f:
                exploits.append(json.load(f))
        return exploits

    def _build_index(self):
        texts = [self._exploit_to_text(e) for e in self.exploits]
        embeddings = self.model.encode(texts, normalize_embeddings=True)
        dim = embeddings.shape[1]
        index = faiss.IndexFlatIP(dim)  # Inner product = cosine similarity khi Д‘ГЈ normalize
        index.add(embeddings.astype(np.float32))
        return index, embeddings

    def retrieve(self, query, k=5):
        query_emb = self.model.encode([query], normalize_embeddings=True)
        scores, indices = self.index.search(query_emb.astype(np.float32), k)
        return [(self.exploits[i], scores[0][j]) for j, i in enumerate(indices[0])]

    def _exploit_to_text(self, exploit):
        return (f"Bridge: {exploit['bridge']}. "
                f"Vulnerability: {exploit['vulnerability_class']}. "
                f"Stage: {exploit['attack_stage']}. "
                f"Trace: {' '.join(exploit['attack_trace'])}. "
                f"Root cause: {exploit['root_cause']}")

struct GlobalSnapshot {
    evm_s_state: EvmState,
    evm_d_state: EvmState,
    relay_state: RelayState,
    timestamp: u64,
}

impl DualEvmManager {
    fn capture_snapshot(&self) -> GlobalSnapshot {
        GlobalSnapshot {
            evm_s_state: self.evm_s.snapshot(),
            evm_d_state: self.evm_d.snapshot(),
            relay_state: self.relay.snapshot(),
            timestamp: std::time::SystemTime::now()
                .duration_since(UNIX_EPOCH)
                .unwrap()
                .as_secs(),
        }
    }

    fn restore_snapshot(&mut self, snap: &GlobalSnapshot) {
        self.evm_s.restore(&snap.evm_s_state);
        self.evm_d.restore(&snap.evm_d_state);
        self.relay.restore(&snap.relay_state);
    }
}

def reward_function(state, waypoints, invariants, alpha=0.3, beta=0.4, gamma=0.3):
    """
    R(sigma) = alpha * cov(sigma) + beta * sum(waypoints_reached) + gamma * inv_dist(sigma)
    """
    # Дҗб»ҷ phб»§ code (basic blocks mб»ӣi trГӘn cбәЈ hai EVM)
    coverage_reward = count_new_basic_blocks(state) / total_basic_blocks

    # Waypoints Д‘ГЈ Д‘бәЎt
    waypoint_reward = sum(
        1 for wp in waypoints if wp.evaluate(state)
    ) / len(waypoints)

    # KhoбәЈng cГЎch Д‘бәҝn vi phбәЎm bбәҘt biбәҝn (0 = vi phбәЎm, 1 = xa nhбәҘt)
    inv_distances = [inv.distance(state) for inv in invariants]
    inv_reward = 1.0 - min(inv_distances) if inv_distances else 0

    return alpha * coverage_reward + beta * waypoint_reward + gamma * inv_reward

8.3 Thu Thбәӯp Metrics

Script thu thбәӯp kбәҝt quбәЈ:

# scripts/collect_results.py
import json
from pathlib import Path

def collect_metrics(results_dir):
    metrics = {}
    for bench_dir in sorted(Path(results_dir).iterdir()):
        if not bench_dir.is_dir():
            continue
        name = bench_dir.name
        runs = []
        for run_file in sorted(bench_dir.glob("run_*.json")):
            with open(run_file) as f:
                runs.append(json.load(f))

        # TГӯnh metrics trung bГ¬nh qua 5 runs
        detected = any(r["vulnerability_found"] for r in runs)
        tte_values = [r["time_to_exploit"] for r in runs if r["vulnerability_found"]]
        median_tte = sorted(tte_values)[len(tte_values)//2] if tte_values else None
        fpr = sum(r["false_positives"] for r in runs) / max(sum(r["total_reports"] for r in runs), 1)

        metrics[name] = {
            "detected": detected,
            "median_tte": median_tte,
            "fpr": fpr,
            "xcc": sum(r["cross_chain_coverage"] for r in runs) / len(runs)
        }
    return metrics
